assignment	: (identifier '=')? expr
expr    	: term (('+'|'-') term)*
term    	: factor (('*'|'/') factor)*
factor		: INT|FLOAT
		: '(' expr ')'
		: identifier
identifier	: ^[a-zA-Z]+$




#### New grammar

statement_list  : (statement ';')+
statement       : exit
                : assignment
assignment      : 'let' identifier type number
exit            : 'exit' number
number          : [0-9]+
identifier      : [a-zA-Z]+[a-zA-Z0-9]+
type            : 'int64'
